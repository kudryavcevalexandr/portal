web/

├─ assets/  # статические ресурсы портала(CSS\JS\иконки), подключаемые на ресурсах портала
│  ├─ css/  # стили портала (общие и для отдельных компонентов\страниц)
│  │  ├─ base.css  # Общие стили (фон, шрифты, кнопки)
│  │  ├─ layout.css  # Сетка, панели, контейнеры
│  │  └─ table.css  # Стили таблиц (в т.ч. дерево-таблица)
│  ├─ img/  # изображения и иконки портала (логотип, svg, png, )
│  └─ js/  # скрипты портала (конфиг, API - клиент, UI-логика, логика страниц)
│     ├─ api.js  # Запросы к Directus (fetch, пагинация)
│     ├─ config.js  # Настройки: Directus URL, коллекции, поля
│     ├─ page-hierarchy-nomen.js  # классификатор позиции на согласование
│     ├─ page-hierarchy.js  # Логика страницы hierarchy.html
│     ├─ page-index.js  # Логика главной страницы (дашборд/отчёты): загрузка данных из os_api (/api/...), отрисовка таблиц/карточек отчётов, кнопка обновления, обработка ошибок
│     ├─ page-pairs-edit.js  # правка позиций - временная
│     ├─ page-search.js  # Логика страницы search.html
│     ├─ page-X.js  # Экспериментальная HTML-страница (прототип / тест)
│     ├─ table-tree.js  # Отрисовка дерева в таблицу + раскрытие веток
│     ├─ tree.js  # Построение дерева из L1/L2/L3/L4
│     └─ ui.js  # Общие UI-хелперы (тосты, лоадер)
├─ nginx/  # Конфигурация веб-сервера Nginx для портала
│  └─ default.conf  # конфигурация Nginx для портала (static, маршрутизация, прокси на API)
├─ pages/  # HTML-страницы портала
│  ├─ about.html  # (пример будущей страницы)
│  ├─ hierarchy-nomen.html  # построение дерева классификатора
│  ├─ hierarchy-nomen-comment.html  # копия hierarchy-nomen для отдельного редактирования
│  ├─ hierarchy.html  # Страница: иерархичная таблица L1→L4
│  ├─ pairs_edit.html  # правка позиций - временная
│  ├─ search.html  # Страница: поиск (отдельно)
│  └─ X.html  # Экспериментальная HTML-страница (прототип / тест)
├─ static/  # Статические файлы, отдаваемые напрямую (без обработки)
│  └─ search/  # Статические ресурсы страницы поиска
│     └─ index.html  # Главная (навигация по разделам)
├─ index.html  # Главная (навигация по разделам)
└─ README.md  # Краткая памятка по структуре
